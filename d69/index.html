💡 核心角度计算原理（Math.atan2）

function calculateAngle(p1, p2) { // p1: 起点（如肩膀）, p2: 终点（如手肘） const dx = p2.x - p1.x; const dy = p2.y - p1.y; // Math.atan2(dy, dx) 返回弧度，转换为角度 const radians = Math.atan2(dy, dx); const degrees = radians * (180 / Math.PI); // 调整为CSS旋转坐标系（0度向上，顺时针为正） return degrees + 90; }
